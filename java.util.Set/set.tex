\frametitle{HashSet}
\begin{center}
  \begin{itemize}[<+->]
    \item backed by Hashmap
    \item fail-fast Iterator
    \item Null-Elemente
    \item langsame iteration .. next O(h/n)
  \end{itemize}
\end{center}

\frame{
\frametitle{CopyOnWriteArraySet}
\begin{center}
  \begin{itemize}[<+->]
    \item nutzt CopyOnWriteArrayList, wiederum verwendet immutable array
    \item bei Änderungen muss ein neues Array erstellt werden
    \item thread-safe
    \item schnelle Leseoperationen, teure Modifizierung
    \item iterator erlaubt keine Modfizierung
    \item Null-Elemente
    \item Iterator fail-safe? TODO
  \end{itemize}
\end{center}
}
